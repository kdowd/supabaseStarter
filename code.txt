    <script type="module">
        // Import from CDN
        import { createClient } from 'https://cdn.jsdelivr.net/npm/@supabase/supabase-js/+esm';

        const supabaseUrl = 'YOUR_SUPABASE_URL'; // Replace with your Supabase URL
        const supabaseAnonKey = 'YOUR_SUPABASE_ANON_KEY'; // Replace with your Supabase Anon Key

        const supabase = createClient(supabaseUrl, supabaseAnonKey);

        console.log('Supabase client created via CDN:', supabase);

        // Example usage:
        async function fetchData() {
            const { data, error } = await supabase.from('your_table_name').select('*');
            if (error) {
                console.error('Error fetching data:', error.message);
            } else {
                console.log('Data:', data);
            }
        }
        fetchData();
    </script>

        import { supabase } from "./js/main.js";
        document.querySelector('button').addEventListener("click", async (e) => {
            console.log("Get Users");
            const { data, error } = await supabase
                .from('titanic')
                .select('*')
                .ilike('country', 'netherlands') // returns array , even if just 1 or 0 results
                .order('first_name', { ascending: true })
                .limit(10);




            if (error) {
                console.log(error);
            }

            if (data) {
                console.log(data);
                //document.querySelector('.results').innerHTML = data;

                    <p>Email: ${element.email}</p>
                    <p>Country: ${element.country}</p>
                    <hr/>`;
                    document.querySelector('.results').appendChild(div);
                });
